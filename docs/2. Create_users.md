# Create users

```shell
## Create users domain
openstack domain create --description "Domain for regular users" users

# Now create operator user in users domain
openstack user create --domain users --password-prompt operator

# Create operator project in users domain
openstack project create --domain users operator-project

# Give operator admin role on their project
openstack role add --user operator --user-domain users \
  --project operator-project --project-domain users admin

openstack role add --user operator --user-domain users \
  --project operator-project --project-domain users member

# create openrc file
cat > ~/operator-openrc << EOF
export OS_AUTH_URL=http://172.16.1.204/openstack-keystone/v3
export OS_USERNAME=operator
export OS_PROJECT_NAME=operator-project
export OS_USER_DOMAIN_NAME=users
export OS_PROJECT_DOMAIN_NAME=users
export OS_IDENTITY_API_VERSION=3
export OS_REGION_NAME=RegionOne
export OS_INTERFACE=public
echo "Please enter OpenStack Password for operator: "
read -sr OS_PASSWORD_INPUT
export OS_PASSWORD=\$OS_PASSWORD_INPUT
EOF
```

## login
```shell
source ~/operator-openrc
```
