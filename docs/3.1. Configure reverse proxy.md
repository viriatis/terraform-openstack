## Access openstack from LAN 
Configured nginx to be able to access openstack from same LAN

```bash
viriatis@viriatis-pc:~$ sudo vim /etc/nginx/sites-available/openstack
server {
    listen 192.168.200.102:80;
    server_name 192.168.200.102;

    set $public_ip "192.168.200.102";
    set $openstack_backend "http://172.16.1.205";
    set $rgw_backend "http://10.57.123.1";

    location / {
        proxy_pass $openstack_backend;
        proxy_set_header Host $public_ip;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header X-Forwarded-Host $public_ip;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_read_timeout 86400;
    }

    location /swift/ {
        proxy_pass $rgw_backend/swift/;
        proxy_set_header Host $public_ip;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}

server {
    listen 3000;
    server_name 192.168.200.102;

    set $rgw_backend "http://10.57.123.1";

    location / {
        proxy_pass $rgw_backend;
        proxy_set_header Host $http_host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}

viriatis@viriatis-pc:~$ sudo rm /etc/nginx/sites-enabled/default
viriatis@viriatis-pc:~$ sudo ln -s /etc/nginx/sites-available/openstack /etc/nginx/sites-enabled/
viriatis@viriatis-pc:~$ sudo nginx -t && sudo systemctl reload nginx
```

## Automated
```bash
PUBLIC_IP=$(ip route get 1.1.1.1 | grep -oP 'src \K[\d.]+')
OPENSTACK_BACKEND=$(sudo k8s kubectl get svc -n openstack traefik-public-lb -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
RGW_BACKEND=$(sudo microceph.ceph config get client.radosgw.gateway rgw_frontends | grep -oP 'endpoint=\K[\d.]+')

sudo tee /etc/nginx/sites-available/openstack << EOF
server {
    listen ${PUBLIC_IP}:80;
    server_name ${PUBLIC_IP};

    set \$public_ip "${PUBLIC_IP}";
    set \$openstack_backend "http://${OPENSTACK_BACKEND}";
    set \$rgw_backend "http://${RGW_BACKEND}";

    location / {
        proxy_pass \$openstack_backend;
        proxy_set_header Host \$public_ip;
        proxy_set_header X-Real-IP \$remote_addr;
        proxy_set_header X-Forwarded-For \$proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto \$scheme;
        proxy_set_header X-Forwarded-Host \$public_ip;
        proxy_http_version 1.1;
        proxy_set_header Upgrade \$http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_read_timeout 86400;
    }

    location /swift/ {
        proxy_pass \$rgw_backend/swift/;
        proxy_set_header Host \$public_ip;
        proxy_set_header X-Real-IP \$remote_addr;
        proxy_set_header X-Forwarded-For \$proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto \$scheme;
    }
}

server {
    listen 3000;
    server_name ${PUBLIC_IP};

    set \$rgw_backend "http://${RGW_BACKEND}";

    location / {
        proxy_pass \$rgw_backend;

}
EOF
sudo rm /etc/nginx/sites-enabled/default
sudo ln -s /etc/nginx/sites-available/openstack /etc/nginx/sites-enabled/
sudo nginx -t && sudo systemctl reload nginx
```
